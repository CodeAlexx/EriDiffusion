// File: candle-kernels/src/group_norm.cu
#include <cuda_fp16.h>
#include <cuda_bf16.h>

#define WARP_SIZE 32
#define MAX_BLOCK_SIZE 1024

// Helper for warp-level reductions
template<typename T>
__device__ T warp_reduce_sum(T val) {
    for (int offset = WARP_SIZE / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(0xffffffff, val, offset);
    }
    return val;
}

template<typename T>
__device__ T block_reduce_sum(T val) {
    __shared__ T shared[32]; // Assuming max 32 warps per block
    int lane = threadIdx.x % WARP_SIZE;
    int wid = threadIdx.x / WARP_SIZE;
    
    val = warp_reduce_sum(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    if (threadIdx.x < blockDim.x / WARP_SIZE) {
        val = shared[threadIdx.x];
    } else {
        val = 0;
    }
    
    if (wid == 0) val = warp_reduce_sum(val);
    
    return val;
}

// Forward pass for GroupNorm
template<typename T>
__global__ void group_norm_forward_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    float* __restrict__ mean,
    float* __restrict__ rstd,
    int batch_size,
    int num_channels,
    int num_groups,
    int spatial_size,
    float eps
) {
    int channels_per_group = num_channels / num_groups;
    
    // Grid: (batch_size * num_groups, 1, 1)
    // Block: (min(spatial_size * channels_per_group, MAX_BLOCK_SIZE), 1, 1)
    int batch_group_idx = blockIdx.x;
    int batch_idx = batch_group_idx / num_groups;
    int group_idx = batch_group_idx % num_groups;
    
    int group_size = channels_per_group * spatial_size;
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    // Calculate mean
    float sum = 0.0f;
    for (int i = tid; i < group_size; i += stride) {
        int c = (i / spatial_size) + group_idx * channels_per_group;
        int s = i % spatial_size;
        int idx = batch_idx * num_channels * spatial_size + c * spatial_size + s;
        sum += (float)input[idx];
    }
    
    sum = block_reduce_sum(sum);
    __shared__ float shared_mean;
    if (tid == 0) {
        shared_mean = sum / group_size;
        mean[batch_group_idx] = shared_mean;
    }
    __syncthreads();
    float group_mean = shared_mean;
    
    // Calculate variance
    float var_sum = 0.0f;
    for (int i = tid; i < group_size; i += stride) {
        int c = (i / spatial_size) + group_idx * channels_per_group;
        int s = i % spatial_size;
        int idx = batch_idx * num_channels * spatial_size + c * spatial_size + s;
        float diff = (float)input[idx] - group_mean;
        var_sum += diff * diff;
    }
    
    var_sum = block_reduce_sum(var_sum);
    __shared__ float shared_rstd;
    if (tid == 0) {
        float var = var_sum / group_size;
        shared_rstd = rsqrtf(var + eps);
        rstd[batch_group_idx] = shared_rstd;
    }
    __syncthreads();
    float group_rstd = shared_rstd;
    
    // Normalize and apply affine transform
    for (int i = tid; i < group_size; i += stride) {
        int c = (i / spatial_size) + group_idx * channels_per_group;
        int s = i % spatial_size;
        int idx = batch_idx * num_channels * spatial_size + c * spatial_size + s;
        
        float normalized = ((float)input[idx] - group_mean) * group_rstd;
        float scaled = normalized;
        
        if (weight != nullptr) {
            scaled *= (float)weight[c];
        }
        if (bias != nullptr) {
            scaled += (float)bias[c];
        }
        
        output[idx] = (T)scaled;
    }
}

// Backward pass for GroupNorm
template<typename T>
__global__ void group_norm_backward_kernel(
    const T* __restrict__ grad_output,
    const T* __restrict__ input,
    const float* __restrict__ mean,
    const float* __restrict__ rstd,
    const T* __restrict__ weight,
    T* __restrict__ grad_input,
    T* __restrict__ grad_weight,
    T* __restrict__ grad_bias,
    int batch_size,
    int num_channels,
    int num_groups,
    int spatial_size
) {
    int channels_per_group = num_channels / num_groups;
    int batch_group_idx = blockIdx.x;
    int batch_idx = batch_group_idx / num_groups;
    int group_idx = batch_group_idx % num_groups;
    
    int group_size = channels_per_group * spatial_size;
    int tid = threadIdx.x;
    int stride = blockDim.x;
    
    float group_mean = mean[batch_group_idx];
    float group_rstd = rstd[batch_group_idx];
    
    // Compute gradients for the group
    float grad_sum = 0.0f;
    float dot_prod = 0.0f;
    
    for (int i = tid; i < group_size; i += stride) {
        int c = (i / spatial_size) + group_idx * channels_per_group;
        int s = i % spatial_size;
        int idx = batch_idx * num_channels * spatial_size + c * spatial_size + s;
        
        float grad_out = (float)grad_output[idx];
        if (weight != nullptr) {
            grad_out *= (float)weight[c];
        }
        
        grad_sum += grad_out;
        float normalized = ((float)input[idx] - group_mean) * group_rstd;
        dot_prod += grad_out * normalized;
    }
    
    grad_sum = block_reduce_sum(grad_sum);
    dot_prod = block_reduce_sum(dot_prod);
    
    __shared__ float shared_grad_sum, shared_dot_prod;
    if (tid == 0) {
        shared_grad_sum = grad_sum;
        shared_dot_prod = dot_prod;
    }
    __syncthreads();
    
    float scale = 1.0f / group_size;
    
    // Compute grad_input
    for (int i = tid; i < group_size; i += stride) {
        int c = (i / spatial_size) + group_idx * channels_per_group;
        int s = i % spatial_size;
        int idx = batch_idx * num_channels * spatial_size + c * spatial_size + s;
        
        float grad_out = (float)grad_output[idx];
        if (weight != nullptr) {
            grad_out *= (float)weight[c];
        }
        
        float normalized = ((float)input[idx] - group_mean) * group_rstd;
        float grad_in = group_rstd * (grad_out - scale * (shared_grad_sum + normalized * shared_dot_prod));
        
        grad_input[idx] = (T)grad_in;
    }
    
    // Compute grad_weight and grad_bias (if needed)
    if ((grad_weight != nullptr || grad_bias != nullptr) && batch_idx == 0) {
        for (int c = tid; c < num_channels; c += stride) {
            if (c >= group_idx * channels_per_group && c < (group_idx + 1) * channels_per_group) {
                float weight_grad_sum = 0.0f;
                float bias_grad_sum = 0.0f;
                
                for (int b = 0; b < batch_size; b++) {
                    for (int s = 0; s < spatial_size; s++) {
                        int idx = b * num_channels * spatial_size + c * spatial_size + s;
                        float grad_out = (float)grad_output[idx];
                        
                        if (grad_bias != nullptr) {
                            bias_grad_sum += grad_out;
                        }
                        
                        if (grad_weight != nullptr) {
                            int bg_idx = b * num_groups + group_idx;
                            float norm_val = ((float)input[idx] - mean[bg_idx]) * rstd[bg_idx];
                            weight_grad_sum += grad_out * norm_val;
                        }
                    }
                }
                
                if (grad_weight != nullptr) {
                    atomicAdd(&grad_weight[c], (T)weight_grad_sum);
                }
                if (grad_bias != nullptr) {
                    atomicAdd(&grad_bias[c], (T)bias_grad_sum);
                }
            }
        }
    }
}

// C interface for Candle
extern "C" {
    void group_norm_forward_f32(
        const float* input,
        const float* weight,
        const float* bias,
        float* output,
        float* mean,
        float* rstd,
        int batch_size,
        int num_channels,
        int num_groups,
        int spatial_size,
        float eps
    ) {
        int total_groups = batch_size * num_groups;
        int channels_per_group = num_channels / num_groups;
        int threads = min(spatial_size * channels_per_group, MAX_BLOCK_SIZE);
        
        group_norm_forward_kernel<<<total_groups, threads>>>(
            input, weight, bias, output, mean, rstd,
            batch_size, num_channels, num_groups, spatial_size, eps
        );
    }
    
    void group_norm_backward_f32(
        const float* grad_output,
        const float* input,
        const float* mean,
        const float* rstd,
        const float* weight,
        float* grad_input,
        float* grad_weight,
        float* grad_bias,
        int batch_size,
        int num_channels,
        int num_groups,
        int spatial_size
    ) {
        int total_groups = batch_size * num_groups;
        int channels_per_group = num_channels / num_groups;
        int threads = min(spatial_size * channels_per_group, MAX_BLOCK_SIZE);
        
        // Zero out gradients for weight and bias
        if (grad_weight != nullptr) {
            cudaMemset(grad_weight, 0, num_channels * sizeof(float));
        }
        if (grad_bias != nullptr) {
            cudaMemset(grad_bias, 0, num_channels * sizeof(float));
        }
        
        group_norm_backward_kernel<<<total_groups, threads>>>(
            grad_output, input, mean, rstd, weight,
            grad_input, grad_weight, grad_bias,
            batch_size, num_channels, num_groups, spatial_size
        );
    }
}

#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cmath>

// RoPE for Flux - handles both 1D (text) and 2D (image) positions
template<typename T>
__global__ void rope_forward_kernel(
    const T* __restrict__ input,  // (batch, seq_len, num_heads, head_dim)
    const int* __restrict__ positions,  // (batch, seq_len) or (batch, seq_len, 2) for 2D
    T* __restrict__ output,
    int batch_size,
    int seq_len,
    int num_heads,
    int head_dim,
    int rotary_dim,  // How many dimensions to rotate (usually head_dim/2 or head_dim)
    float theta_base,
    bool is_2d  // true for image positions, false for text
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * seq_len * num_heads * head_dim;
    
    if (idx >= total_elements) return;
    
    // Decompose index
    int b = idx / (seq_len * num_heads * head_dim);
    int s = (idx / (num_heads * head_dim)) % seq_len;
    int h = (idx / head_dim) % num_heads;
    int d = idx % head_dim;
    
    // Copy non-rotated dimensions directly
    if (d >= rotary_dim) {
        output[idx] = input[idx];
        return;
    }
    
    // Get position
    float pos;
    if (is_2d && d < rotary_dim / 2) {
        // First half for height position
        pos = (float)positions[b * seq_len * 2 + s * 2];
    } else if (is_2d) {
        // Second half for width position
        pos = (float)positions[b * seq_len * 2 + s * 2 + 1];
        d = d - rotary_dim / 2;  // Adjust dimension index
    } else {
        // 1D position
        pos = (float)positions[b * seq_len + s];
    }
    
    // RoPE calculation
    int d_pair = d / 2;
    float freq = 1.0f / powf(theta_base, 2.0f * d_pair / (float)rotary_dim);
    float angle = pos * freq;
    float cos_angle = cosf(angle);
    float sin_angle = sinf(angle);
    
    // Get the pair of values to rotate
    int pair_idx = (d % 2 == 0) ? idx : idx - 1;
    float x0 = (float)input[pair_idx];
    float x1 = (float)input[pair_idx + 1];
    
    // Apply rotation
    if (d % 2 == 0) {
        output[idx] = (T)(x0 * cos_angle - x1 * sin_angle);
    } else {
        output[idx] = (T)(x0 * sin_angle + x1 * cos_angle);
    }
}

// Optimized version with precomputed cos/sin cache
template<typename T>
__global__ void rope_forward_cached_kernel(
    const T* __restrict__ input,
    const float* __restrict__ cos_cache,  // (max_seq_len, rotary_dim/2)
    const float* __restrict__ sin_cache,  // (max_seq_len, rotary_dim/2)
    const int* __restrict__ positions,
    T* __restrict__ output,
    int batch_size,
    int seq_len,
    int num_heads,
    int head_dim,
    int rotary_dim,
    int max_cached_len,
    bool is_2d
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * seq_len * num_heads * head_dim;
    
    if (idx >= total_elements) return;
    
    int b = idx / (seq_len * num_heads * head_dim);
    int s = (idx / (num_heads * head_dim)) % seq_len;
    int h = (idx / head_dim) % num_heads;
    int d = idx % head_dim;
    
    if (d >= rotary_dim) {
        output[idx] = input[idx];
        return;
    }
    
    int pos_idx;
    int d_adj = d;
    
    if (is_2d && d < rotary_dim / 2) {
        pos_idx = positions[b * seq_len * 2 + s * 2];
    } else if (is_2d) {
        pos_idx = positions[b * seq_len * 2 + s * 2 + 1];
        d_adj = d - rotary_dim / 2;
    } else {
        pos_idx = positions[b * seq_len + s];
    }
    
    // Ensure position is within cache bounds
    if (pos_idx >= max_cached_len) {
        // Fallback to computing on the fly
        output[idx] = input[idx];  // Or compute RoPE dynamically
        return;
    }
    
    int d_pair = d_adj / 2;
    int cache_idx = pos_idx * (rotary_dim / 2) + d_pair;
    
    float cos_val = cos_cache[cache_idx];
    float sin_val = sin_cache[cache_idx];
    
    int pair_idx = (d % 2 == 0) ? idx : idx - 1;
    float x0 = (float)input[pair_idx];
    float x1 = (float)input[pair_idx + 1];
    
    if (d % 2 == 0) {
        output[idx] = (T)(x0 * cos_val - x1 * sin_val);
    } else {
        output[idx] = (T)(x0 * sin_val + x1 * cos_val);
    }
}

// Precompute cos/sin cache
__global__ void precompute_rope_cache_kernel(
    float* cos_cache,
    float* sin_cache,
    int max_seq_len,
    int rotary_dim,
    float theta_base
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = max_seq_len * (rotary_dim / 2);
    
    if (idx >= total) return;
    
    int pos = idx / (rotary_dim / 2);
    int d_pair = idx % (rotary_dim / 2);
    
    float freq = 1.0f / powf(theta_base, 2.0f * d_pair / (float)rotary_dim);
    float angle = pos * freq;
    
    cos_cache[idx] = cosf(angle);
    sin_cache[idx] = sinf(angle);
}

extern "C" {

use candle::{CpuStorage, CustomOp1, CustomOp2, Layout, Result, Shape, Tensor, D};
use candle_cuda_kernels::group_norm as cuda_kernels;

pub struct GroupNorm {
    pub num_groups: usize,
    pub eps: f64,
    pub affine: bool,
}

impl GroupNorm {
    pub fn new(num_groups: usize, eps: f64, affine: bool) -> Self {
        Self {
            num_groups,
            eps,
            affine,
        }
    }

    pub fn forward(
        &self,
        x: &Tensor,
        weight: Option<&Tensor>,
        bias: Option<&Tensor>,
    ) -> Result<(Tensor, Tensor, Tensor)> {
        let device = x.device();
        let dtype = x.dtype();
        
        // Validate inputs
        let (b, c, h, w) = x.dims4()?;
        if c % self.num_groups != 0 {
            candle::bail!("num_channels must be divisible by num_groups");
        }

        // Reshape for group norm: (B, C, H, W) -> (B, C, H*W)
        let x = x.reshape(&[b, c, h * w])?;
        let spatial_size = h * w;

        // Prepare outputs
        let mean = Tensor::zeros(&[b * self.num_groups], dtype, device)?;
        let rstd = Tensor::zeros(&[b * self.num_groups], dtype, device)?;
        let output = Tensor::zeros_like(&x)?;

        // Call CUDA kernel
        match device {
            candle::Device::Cuda(cuda_device) => {
                unsafe {
                    let cuda_x = x.as_cuda_slice::<f32>()?;
                    let cuda_weight = weight.map(|w| w.as_cuda_slice::<f32>()).transpose()?;
                    let cuda_bias = bias.map(|b| b.as_cuda_slice::<f32>()).transpose()?;
                    let cuda_output = output.as_cuda_slice_mut::<f32>()?;
                    let cuda_mean = mean.as_cuda_slice_mut::<f32>()?;
                    let cuda_rstd = rstd.as_cuda_slice_mut::<f32>()?;

                    cuda_kernels::group_norm_forward_f32(
                        cuda_x.as_ptr(),
                        cuda_weight.map(|w| w.as_ptr()).unwrap_or(std::ptr::null()),
                        cuda_bias.map(|b| b.as_ptr()).unwrap_or(std::ptr::null()),
                        cuda_output.as_mut_ptr(),
                        cuda_mean.as_mut_ptr(),
                        cuda_rstd.as_mut_ptr(),
                        b as i32,
                        c as i32,
                        self.num_groups as i32,
                        spatial_size as i32,
                        self.eps as f32,
                    );
                }
            }
            _ => {
                // CPU fallback implementation
                candle::bail!("GroupNorm CPU implementation not yet available");
            }
        }

        // Reshape output back
        let output = output.reshape(&[b, c, h, w])?;
        
        Ok((output, mean, rstd))
    }
}

use candle::{Device, Result, Tensor};

pub struct RotaryEmbedding {
    max_seq_len: usize,
    dim: usize,
    base: f32,
    cached_cos: Option<Tensor>,
    cached_sin: Option<Tensor>,
}

impl RotaryEmbedding {
    pub fn new(dim: usize, max_seq_len: usize, base: f32, device: &Device) -> Result<Self> {
        let mut rope = Self {
            max_seq_len,
            dim,
            base,
            cached_cos: None,
            cached_sin: None,
        };
        
        rope.precompute_cache(device)?;
        Ok(rope)
    }

    fn precompute_cache(&mut self, device: &Device) -> Result<()> {
        let rotary_dim = self.dim;
        let cache_size = self.max_seq_len * (rotary_dim / 2);
        
        let cos_cache = Tensor::zeros(cache_size, candle::DType::F32, device)?;
        let sin_cache = Tensor::zeros(cache_size, candle::DType::F32, device)?;

        match device {
            Device::Cuda(_) => {
                unsafe {
                    let cos_ptr = cos_cache.as_cuda_slice_mut::<f32>()?.as_mut_ptr();
                    let sin_ptr = sin_cache.as_cuda_slice_mut::<f32>()?.as_mut_ptr();
                    
                    candle_cuda_kernels::rope::precompute_rope_cache_f32(
                        cos_ptr,
                        sin_ptr,
                        self.max_seq_len as i32,
                        rotary_dim as i32,
                        self.base,
                    );
                }
            }
            _ => {
                // CPU implementation
                candle::bail!("RoPE CPU cache precomputation not implemented");
            }
        }

        self.cached_cos = Some(cos_cache.reshape(&[self.max_seq_len, rotary_dim / 2])?);
        self.cached_sin = Some(sin_cache.reshape(&[self.max_seq_len, rotary_dim / 2])?);
        
        Ok(())
    }

    pub fn forward(
        &self,
        x: &Tensor,
        positions: &Tensor,
        is_2d: bool,
    ) -> Result<Tensor> {
        let device = x.device();
        let (batch_size, seq_len, num_heads, head_dim) = x.dims4()?;
        
        let output = Tensor::zeros_like(x)?;

        match device {
            Device::Cuda(_) => {
                unsafe {
                    let x_ptr = x.as_cuda_slice::<f32>()?.as_ptr();
                    let pos_ptr = positions.as_cuda_slice::<i32>()?.as_ptr();
                    let out_ptr = output.as_cuda_slice_mut::<f32>()?.as_mut_ptr();
                    
                    if let (Some(cos), Some(sin)) = (&self.cached_cos, &self.cached_sin) {
                        let cos_ptr = cos.as_cuda_slice::<f32>()?.as_ptr();
                        let sin_ptr = sin.as_cuda_slice::<f32>()?.as_ptr();
                        
                        candle_cuda_kernels::rope::rope_forward_cached_f32(
                            x_ptr,
                            cos_ptr,
                            sin_ptr,
                            pos_ptr,
                            out_ptr,
                            batch_size as i32,
                            seq_len as i32,
                            num_heads as i32,
                            head_dim as i32,
                            self.dim as i32,
                            self.max_seq_len as i32,
                            is_2d,
                        );
                    } else {
                        candle_cuda_kernels::rope::rope_forward_f32(
                            x_ptr,
                            pos_ptr,
                            out_ptr,
                            batch_size as i32,

// File: candle-kernels/src/group_norm.cu
#include <cuda_fp16.h>
#include <cuda_bf16.h>
#include <cooperative_groups.h>
#include <cuda/std/type_traits>

namespace cg = cooperative_groups;

#define WARP_SIZE 32
#define MAX_THREADS_PER_BLOCK 256

// Helper for type dispatch
template<typename T>
struct AccType {
    using type = float;
};

// Warp-level reduction with proper sync
template<typename T>
__device__ T warp_reduce_sum(T val) {
    unsigned mask = __activemask();
    for (int offset = WARP_SIZE / 2; offset > 0; offset /= 2) {
        val += __shfl_down_sync(mask, val, offset);
    }
    return val;
}

// Block-level reduction
template<typename T>
__device__ T block_reduce_sum(T val, T* shared) {
    int lane = threadIdx.x % WARP_SIZE;
    int wid = threadIdx.x / WARP_SIZE;
    
    val = warp_reduce_sum(val);
    
    if (lane == 0) shared[wid] = val;
    __syncthreads();
    
    // Final warp reduction
    if (threadIdx.x < blockDim.x / WARP_SIZE) {
        val = shared[threadIdx.x];
    } else {
        val = 0;
    }
    
    if (wid == 0) val = warp_reduce_sum(val);
    
    return val;
}

// Fixed GroupNorm forward kernel
template<typename T>
__global__ void group_norm_forward_kernel(
    const T* __restrict__ input,
    const T* __restrict__ weight,
    const T* __restrict__ bias,
    T* __restrict__ output,
    float* __restrict__ mean_out,
    float* __restrict__ rstd_out,
    int batch_size,
    int num_channels,
    int num_groups,
    int spatial_size,
    float eps
) {
    using AccT = typename AccType<T>::type;
    
    // Grid: (batch_size, num_groups)
    // Block: (min(spatial_size, MAX_THREADS_PER_BLOCK))
    int batch_idx = blockIdx.x;
    int group_idx = blockIdx.y;
    
    if (batch_idx >= batch_size || group_idx >= num_groups) return;
    
    int channels_per_group = num_channels / num_groups;
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    // Shared memory for reductions
    extern __shared__ char shared_mem[];
    AccT* shared_sum = (AccT*)shared_mem;
    AccT* shared_var = (AccT*)(shared_mem + block_size * sizeof(AccT));
    
    // Calculate mean - grid-stride loop over spatial*channel dims
    AccT sum = 0;
    for (int i = tid; i < channels_per_group * spatial_size; i += block_size) {
        int c = i / spatial_size;
        int s = i % spatial_size;
        int channel_idx = group_idx * channels_per_group + c;
        int idx = ((batch_idx * num_channels + channel_idx) * spatial_size) + s;
        sum += (AccT)input[idx];
    }
    
    sum = block_reduce_sum(sum, shared_sum);
    
    __shared__ AccT group_mean;
    if (tid == 0) {
        group_mean = sum / (channels_per_group * spatial_size);
        mean_out[batch_idx * num_groups + group_idx] = group_mean;
    }
    __syncthreads();
    
    // Calculate variance
    AccT var_sum = 0;
    for (int i = tid; i < channels_per_group * spatial_size; i += block_size) {
        int c = i / spatial_size;
        int s = i % spatial_size;
        int channel_idx = group_idx * channels_per_group + c;
        int idx = ((batch_idx * num_channels + channel_idx) * spatial_size) + s;
        AccT diff = (AccT)input[idx] - group_mean;
        var_sum += diff * diff;
    }
    
    var_sum = block_reduce_sum(var_sum, shared_var);
    
    __shared__ AccT group_rstd;
    if (tid == 0) {
        AccT var = var_sum / (channels_per_group * spatial_size);
        group_rstd = rsqrtf(var + eps);
        rstd_out[batch_idx * num_groups + group_idx] = group_rstd;
    }
    __syncthreads();
    
    // Normalize and apply affine transform
    for (int i = tid; i < channels_per_group * spatial_size; i += block_size) {
        int c = i / spatial_size;
        int s = i % spatial_size;
        int channel_idx = group_idx * channels_per_group + c;
        int idx = ((batch_idx * num_channels + channel_idx) * spatial_size) + s;
        
        AccT normalized = ((AccT)input[idx] - group_mean) * group_rstd;
        
        if (weight != nullptr) {
            normalized *= (AccT)weight[channel_idx];
        }
        if (bias != nullptr) {
            normalized += (AccT)bias[channel_idx];
        }
        
        output[idx] = (T)normalized;
    }
}

// Fixed backward kernel with proper synchronization
template<typename T>
__global__ void group_norm_backward_kernel(
    const T* __restrict__ grad_output,
    const T* __restrict__ input,
    const float* __restrict__ mean,
    const float* __restrict__ rstd,
    const T* __restrict__ weight,
    T* __restrict__ grad_input,
    T* __restrict__ grad_weight,
    T* __restrict__ grad_bias,
    int batch_size,
    int num_channels,
    int num_groups,
    int spatial_size
) {
    using AccT = typename AccType<T>::type;
    
    int batch_idx = blockIdx.x;
    int group_idx = blockIdx.y;
    
    if (batch_idx >= batch_size || group_idx >= num_groups) return;
    
    int channels_per_group = num_channels / num_groups;
    int tid = threadIdx.x;
    int block_size = blockDim.x;
    
    extern __shared__ char shared_mem[];
    AccT* shared_sum = (AccT*)shared_mem;
    AccT* shared_dot = (AccT*)(shared_mem + block_size * sizeof(AccT));
    
    AccT group_mean = mean[batch_idx * num_groups + group_idx];
    AccT group_rstd = rstd[batch_idx * num_groups + group_idx];
    
    // Compute sum(dL/dy) and sum(dL/dy * (x - mean) / std)
    AccT grad_sum = 0;
    AccT dot_prod = 0;
    
    for (int i = tid; i < channels_per_group * spatial_size; i += block_size) {
        int c = i / spatial_size;
        int s = i % spatial_size;
        int channel_idx = group_idx * channels_per_group + c;
        int idx = ((batch_idx * num_channels + channel_idx) * spatial_size) + s;
        
        AccT grad_out = (AccT)grad_output[idx];
        if (weight != nullptr) {
            grad_out *= (AccT)weight[channel_idx];
        }
        
        grad_sum += grad_out;
        AccT normalized = ((AccT)input[idx] - group_mean) * group_rstd;
        dot_prod += grad_out * normalized;
    }
    
    grad_sum = block_reduce_sum(grad_sum, shared_sum);
    dot_prod = block_reduce_sum(dot_prod, shared_dot);
    
    __shared__ AccT shared_grad_sum, shared_dot_prod;
    if (tid == 0) {
        shared_grad_sum = grad_sum;
        shared_dot_prod = dot_prod;
    }
    __syncthreads();
    
    AccT scale = 1.0f / (channels_per_group * spatial_size);
    
    // Compute grad_input
    for (int i = tid; i < channels_per_group * spatial_size; i += block_size) {
        int c = i / spatial_size;
        int s = i % spatial_size;
        int channel_idx = group_idx * channels_per_group + c;
        int idx = ((batch_idx * num_channels + channel_idx) * spatial_size) + s;
        
        AccT grad_out = (AccT)grad_output[idx];
        if (weight != nullptr) {
            grad_out *= (AccT)weight[channel_idx];
        }
        
        AccT normalized = ((AccT)input[idx] - group_mean) * group_rstd;
        AccT grad_in = group_rstd * (grad_out - scale * (shared_grad_sum + normalized * shared_dot_prod));
        
        grad_input[idx] = (T)grad_in;
    }
    
    // Accumulate gradients for weight and bias (only from first batch for each channel)
    if (batch_idx == 0 && (grad_weight != nullptr || grad_bias != nullptr)) {
        for (int c = tid; c < channels_per_group; c += block_size) {
            int channel_idx = group_idx * channels_per_group + c;
            AccT weight_grad = 0;
            AccT bias_grad = 0;
            
            // Sum over all batches and spatial positions
            for (int b = 0; b < batch_size; b++) {
                int b_mean_idx = b * num_groups + group_idx;
                AccT b_mean = mean[b_mean_idx];
                AccT b_rstd = rstd[b_mean_idx];
                
                for (int s = 0; s < spatial_size; s++) {
                    int idx = ((b * num_channels + channel_idx) * spatial_size) + s;
                    AccT grad_out = (AccT)grad_output[idx];
                    
                    if (grad_bias != nullptr) {
                        bias_grad += grad_out;
                    }
                    
                    if (grad_weight != nullptr) {
                        AccT normalized = ((AccT)input[idx] - b_mean) * b_rstd;
                        weight_grad += grad_out * normalized;
                    }
                }
            }
            
            if (grad_weight != nullptr) {
                atomicAdd(&grad_weight[channel_idx], (T)weight_grad);
            }
            if (grad_bias != nullptr) {
                atomicAdd(&grad_bias[channel_idx], (T)bias_grad);
            }
        }
    }
}

// Fixed RoPE kernel with correct indexing
template<typename T>
__global__ void rope_forward_kernel(
    const T* __restrict__ input,
    const int* __restrict__ positions,
    T* __restrict__ output,
    int batch_size,
    int seq_len,
    int num_heads,
    int head_dim,
    int rotary_dim,
    float theta_base,
    int is_2d  // Changed from bool to int for FFI
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total_elements = batch_size * seq_len * num_heads * head_dim;
    
    if (idx >= total_elements) return;
    
    // Decompose index
    int b = idx / (seq_len * num_heads * head_dim);
    int s = (idx / (num_heads * head_dim)) % seq_len;
    int h = (idx / head_dim) % num_heads;
    int d = idx % head_dim;
    
    // Copy non-rotated dimensions
    if (d >= rotary_dim) {
        output[idx] = input[idx];
        return;
    }
    
    // Ensure we're working with pairs
    if (d % 2 == 1) return;  // Process pairs in even indices only
    
    // Get position
    float pos;
    int d_rot = d;  // Keep original d for indexing
    
    if (is_2d) {
        // For 2D: first half of rotary_dim for height, second half for width
        if (d < rotary_dim / 2) {
            pos = (float)positions[b * seq_len * 2 + s * 2];
        } else {
            pos = (float)positions[b * seq_len * 2 + s * 2 + 1];
            d_rot = d - rotary_dim / 2;
        }
    } else {
        // 1D position
        pos = (float)positions[b * seq_len + s];
    }
    
    // Calculate rotation
    int d_pair = d_rot / 2;
    float freq = 1.0f / powf(theta_base, 2.0f * d_pair / (float)rotary_dim);
    float angle = pos * freq;
    float cos_angle = cosf(angle);
    float sin_angle = sinf(angle);
    
    // Apply rotation to pair
    float x0 = (float)input[idx];
    float x1 = (float)input[idx + 1];
    
    output[idx] = (T)(x0 * cos_angle - x1 * sin_angle);
    output[idx + 1] = (T)(x0 * sin_angle + x1 * cos_angle);
}

template<typename T>
__global__ void rope_forward_cached_kernel(

        return;

        float x1 = (float)input[idx + 1];

        output[idx + 1] = (T)(x0 * sin_val + x1 * cos_val);
        return;
    }
    
    int d_pair = d_rot / 2;
    int cache_idx = pos_idx * cache_dim + d_pair;
    
    float cos_val = cos_cache[cache_idx];
    float sin_val = sin_cache[cache_idx];
    
    float x0 = (float)input[idx];
    float x1 = (float)input[idx + 1];
    
    output[idx] = (T)(x0 * cos_val - x1 * sin_val);
    output[idx + 1] = (T)(x0 * sin_val + x1 * cos_val);
}

// Cache precomputation
__global__ void precompute_rope_cache_kernel(
    float* cos_cache,
    float* sin_cache,
    int max_seq_len,
    int rotary_dim,
    float theta_base
) {
    int idx = blockIdx.x * blockDim.x + threadIdx.x;
    int total = max_seq_len * (rotary_dim / 2);
    
    if (idx >= total) return;
    
    int pos = idx / (rotary_dim / 2);
    int d_pair = idx % (rotary_dim / 2);
    
    float freq = 1.0f / powf(theta_base, 2.0f * d_pair / (float)rotary_dim);
    float angle = pos * freq;
    
    cos_cache[idx] = cosf(angle);
    sin_cache[idx] = sinf(angle);
}

extern "C" {
    cudaError_t rope_forward_f32(
        const float* input,
        const int* positions,
        float* output,
        int batch_size,
        int seq_len,
        int num_heads,
        int head_dim,
        int rotary_dim,
        float theta_base,
        int is_2d
    ) {
        int total = batch_size * seq_len * num_heads * head_dim;
        int threads = 256;
        int blocks = (total + threads - 1) / threads;
        
        rope_forward_kernel<<<blocks, threads>>>(
            input, positions, output,
            batch_size, seq_len, num_heads, head_dim,
            rotary_dim, theta_base, is_2d
        );
        
        return cudaGetLastError();
    }

    ) {

use candle::{CpuStorage, CustomOp1, Layout, Result, Shape, Tensor, DType, Device, WithDType};
use candle_cuda_kernels::group_norm as cuda_kernels;

#[derive(Debug, Clone)]
pub struct GroupNorm {
    pub num_groups: usize,
    pub eps: f64,
    pub affine: bool,
}

impl GroupNorm {
    pub fn new(num_groups: usize, num_channels: usize, eps: f64, affine: bool) -> Result<Self> {
        if num_channels % num_groups != 0 {
            candle::bail!(
                "num_channels ({}) must be divisible by num_groups ({})",
                num_channels,
                num_groups
            );
        }
        Ok(Self {
            num_groups,
            eps,
            affine,
        })
    }

    pub fn forward(
        &self,
        x: &Tensor,
        weight: Option<&Tensor>,
        bias: Option<&Tensor>,
    ) -> Result<(Tensor, Tensor, Tensor)> {
        let device = x.device();
        let dtype = x.dtype();
        
        // Handle different input shapes
        let original_shape = x.shape().clone();
        let x_dims = x.dims();
        
        let (batch_size, num_channels, spatial_size) = match x_dims.len() {
            3 => {
                // (B, C, L) - already in correct format
                let &[b, c, l] = x_dims else { unreachable!() };
                (b, c, l)
            }
            4 => {
                // (B, C, H, W) - reshape to (B, C, H*W)
                let &[b, c, h, w] = x_dims else { unreachable!() };
                (b, c, h * w)
            }
            _ => candle::bail!("GroupNorm expects 3D or 4D input, got {:?}", x_dims),
        };

        // Reshape if needed
        let x = if x_dims.len() == 4 {
            x.reshape(&[batch_size, num_channels, spatial_size])?
        } else {
            x.clone()
        };

        // Validate affine parameters
        if self.affine {
            if let Some(w) = weight {
                if w.dims() != [num_channels] {
                    candle::bail!("weight must have shape [{}], got {:?}", num_channels, w.dims());
                }
            }
            if let Some(b) = bias {
                if b.dims() != [num_channels] {
                    candle::bail!("bias must have shape [{}], got {:?}", num_channels, b.dims());
                }
            }
        }

        // Prepare output tensors
        let mean = Tensor::zeros(&[batch_size * self.num_groups], dtype, device)?;
        let rstd = Tensor::zeros(&[batch_size * self.num_groups], dtype, device)?;
        let output = Tensor::zeros_like(&x)?;

        // Call appropriate kernel based on device and dtype
        match (device, dtype) {
            (Device::Cuda(_), DType::F32) => {
                self.cuda_forward_f32(&x, weight, bias, &output, &mean, &rstd, 
                                     batch_size, num_channels, spatial_size)?;
            }
            (Device::Cuda(_), DType::BF16) => {
                // Convert to f32, compute, convert back
                let x_f32 = x.to_dtype(DType::F32)?;
                let output_f32 = Tensor::zeros_like(&x_f32)?;
                let weight_f32 = weight.map(|w| w.to_dtype(DType::F32)).transpose()?;
                let bias_f32 = bias.map(|b| b.to_dtype(DType::F32)).transpose()?;
                
                self.cuda_forward_f32(&x_f32, weight_f32.as_ref(), bias_f32.as_ref(), 
                                     &output_f32, &mean, &rstd,
                                     batch_size, num_channels, spatial_size)?;
                
                let output = output_f32.to_dtype(DType::BF16)?;
                output.reshape(&original_shape)?;
            }
            (Device::Cpu, _) => {
                candle::bail!("GroupNorm CPU implementation not available");
            }
            _ => {
                candle::bail!("Unsupported device/dtype combination for GroupNorm");
            }
        }

        // Reshape output back to original shape if needed
        let output = if x_dims.len() == 4 {
            output.reshape(&original_shape)?
        } else {
            output
        };

        Ok((output, mean, rstd))
    }

    fn cuda_forward_f32(
        &self,
        x: &Tensor,
        weight: Option<&Tensor>,
        bias: Option<&Tensor>,
        output: &Tensor,
        mean: &Tensor,
        rstd: &Tensor,
        batch_size: usize,
        num_channels: usize,
        spatial_size: usize,
    ) -> Result<()> {
        use candle::cuda_backend::WrapErr;
        
        let device = x.device().as_cuda_device()?;
        
        // Get CUDA storage and ensure they're contiguous
        let x = x.as_cuda_storage()?;
        let output = output.as_cuda_storage()?;
        let mean = mean.as_cuda_storage()?;
        let rstd = rstd.as_cuda_storage()?;
        
        let weight_ptr = weight
            .map(|w| w.as_cuda_storage().map(|s| s.as_ptr()))
            .transpose()?
            .unwrap_or(std::ptr::null());
            
        let bias_ptr = bias
            .map(|b| b.as_cuda_storage().map(|s| s.as_ptr()))
            .transpose()?
            .unwrap_or(std::ptr::null());

        // Call CUDA kernel
        let err = unsafe {
            cuda_kernels::group_norm_forward_f32(
                x.as_ptr(),
                weight_ptr,
                bias_ptr,
                output.as_ptr() as *mut f32,
                mean.as_ptr() as *mut f32,
                rstd.as_ptr() as *mut f32,
                batch_size as i32,
                num_channels as i32,
                self.num_groups as i32,
                spatial_size as i32,
                self.eps as f32,
            )
        };
        
        if err != 0 {
            candle::bail!("CUDA error in group_norm_forward: {}", err);
        }
        
        Ok(())
    }
}

// Layer wrapper for convenience
pub fn group_norm(
    num_groups: usize,
    num_channels: usize,
    eps: f64,
    affine: bool,
    vb: candle_nn::VarBuilder,
) -> Result<impl Fn(&Tensor) -> Result<Tensor>> {
    let group_norm = GroupNorm::new(num_groups, num_channels, eps, affine)?;
    
    let weight = if affine {
        Some(vb.get(num_channels, "weight")?)
    } else {
        None
    };
    
    let bias = if affine {
        Some(vb.get(num_channels, "bias")?)
    } else {
        None
    };
    
    Ok(move |x: &Tensor| {
        let (output, _, _) = group_norm.forward(x, weight.as_ref(), bias.as_ref())?;
        Ok(output)
    })
}

use candle::{Device, Result, Tensor, DType};

#[derive(Debug, Clone)]
pub struct RotaryEmbedding {
    max_seq_len: usize,
    dim: usize,
    base: f32,
    cached_cos: Option<Tensor>,
    cached_sin: Option<Tensor>,
}

impl RotaryEmbedding {
    pub fn new(dim: usize, max_seq_len: usize, base: f32, device: &Device) -> Result<Self> {
        // Ensure dim is even
        if dim % 2 != 0 {
            candle::bail!("RoPE dimension must be even, got {}", dim);
        }
        
        let mut rope = Self {
            max_seq_len,
            dim,
            base,
            cached_cos: None,
            cached_sin: None,
        };
        
        if matches!(device, Device::Cuda(_)) {
            rope.precompute_cache(device)?;
        }
        
        Ok(rope)
    }

    fn precompute_cache(&mut self, device: &Device) -> Result<()> {
        let cache_size = self.max_seq_len * (self.dim / 2);
        
        let cos_cache = Tensor::zeros(&[cache_size], DType::F32, device)?;
        let sin_cache = Tensor::zeros(&[cache_size], DType::F32, device)?;

        match device {
            Device::Cuda(_) => {
                use candle::cuda_backend::WrapErr;
                
                let cos_storage = cos_cache.as_cuda_storage()?;
                let sin_storage = sin_cache.as_cuda_storage()?;
                
                let err = unsafe {
                    candle_cuda_kernels::rope::precompute_rope_cache_f32(
                        cos_storage.as_ptr() as *mut f32,
                        sin_storage.as_ptr() as *mut f32,
                        self.max_seq_len as i32,
                        self.dim as i32,
                        self.base,
                    )
                };
                
                if err != 0 {
                    candle::bail!("CUDA error in precompute_rope_cache: {}", err);
                }
            }
            _ => {
                // CPU implementation would go here
                candle::bail!("RoPE CPU cache precomputation not implemented");
            }
        }

        self.cached_cos = Some(cos_cache.reshape(&[self.max_seq_len, self.dim / 2])?);
        self.cached_sin = Some(sin_cache.reshape(&[self.max_seq_len, self.dim / 2])?);
        
        Ok(())
    }

    pub fn forward(
        &self,
        x: &Tensor,
        positions: &Tensor,
        is_2d: bool,
    ) -> Result<Tensor> {
        let device = x.device();
        let dtype = x.dtype();
        let x_dims = x.dims();
        
        // Validate input shape
        let (batch_size, seq_len, num_heads, head_dim) = match x_dims {
            &[b, s, n, h] => (b, s, n, h),
            &[b, s, h] => (b, s, 1, h),  // Add dummy head dimension
            _ => candle::bail!("RoPE expects input shape [B, S, N, H] or [B, S, H], got {:?}", x_dims),
        };
        
        // Validate positions shape
        let expected_pos_shape = if is_2d {
            vec![batch_size, seq_len, 2]
        } else {
            vec![batch_size, seq_len]
        };
        
        if positions.dims() != expected_pos_shape {
            candle::bail!("Expected positions shape {:?}, got {:?}", expected_pos_shape, positions.dims());
        }
        
        // Ensure positions are i32
        let positions = if positions.dtype() != DType::I32 {
            positions.to_dtype(DType::I32)?
        } else {
            positions.clone()
        };

        // Reshape x if needed
        let x = if x_dims.len() == 3 {
            x.reshape(&[batch_size, seq_len, num_heads, head_dim])?
        } else {
            x.clone()
        };

        let output = match (device, dtype) {
            (Device::Cuda(_), DType::F32) => {
                self.cuda_forward_f32(&x, &positions, batch_size, seq_len, num_heads, head_dim, is_2d)?
            }
            (Device::Cuda(_), DType::BF16) => {
                // Convert to F32, apply RoPE, convert back
                let x_f32 = x.to_dtype(DType::F32)?;
                let output_f32 = self.cuda_forward_f32(&x_f32, &positions, batch_size, seq_len, num_heads, head_dim, is_2d)?;
                output_f32.to_dtype(DType::BF16)?
            }
            _ => candle::bail!("RoPE only supports CUDA device with F32/BF16"),
        };

        // Reshape back if needed
        if x_dims.len() == 3 {
            output.reshape(x_dims)
        } else {
            Ok(output)
        }
    }

    fn cuda_forward_f32(
        &self,
        x: &Tensor,
        positions: &Tensor,
        batch_size: usize,
        seq_len: usize,
        num_heads: usize,
        head_dim: usize,
        is_2d: bool,
    ) -> Result<Tensor> {
        use candle::cuda_backend::WrapErr;
        
        let output = Tensor::zeros_like(x)?;
        
        let x_storage = x.as_cuda_storage()?;
        let pos_storage = positions.as_cuda_storage()?;
        let out_storage = output.as_cuda_storage()?;
        
        let err = if let (Some(cos), Some(sin)) = (&self.cached_cos, &self.cached_sin) {
            let cos_storage = cos.as_cuda_storage()?;
            let sin_storage = sin.as_cuda_storage()?;
            
            unsafe {
                candle_cuda_kernels::rope::rope_forward_cached_f32(
                    x_storage.as_ptr(),
                    cos_storage.as_ptr(),
                    sin_storage.as_ptr(),
                    pos_storage.as_ptr() as *const i32,
                    out_storage.as_ptr() as *mut f32,
                    batch_size as i32,
                    seq_len as i32,
                    num_heads as i32,
                    head_dim as i32,
                    self.dim as i32,
                    self.max_seq_len as i32,
                    if is_2d { 1 } else { 0 },
                )
            }
        } else {
            unsafe {
                candle_cuda_kernels::rope::rope_forward_f32(
                    x_storage.as_ptr(),
                    pos_storage.as_ptr() as *const i32,
                    out_storage.as_ptr() as *mut f32,
                    batch_size as i32,
                    seq_len as i32,
                    num_heads as i32,
                    head_dim as i32,
                    self.dim as i32,
                    self.base,
                    if is_2d { 1 } else { 0 },
                )
            }
        };
        
        if err != 0 {
            candle::bail!("CUDA error in rope_forward: {}", err);
        }
        
        Ok(output)
    }
}

// Helper for Flux 2D positions
pub fn get_2d_positions(height: usize, width: usize, device: &Device) -> Result<Tensor> {
    let mut positions = vec![0i32; height * width * 2];
    
    for h in 0..height {
        for w in 0..width {
            let idx = h * width + w;
            positions[idx * 2] = h as i32;
            positions[idx * 2 + 1] = w as i32;
        }
    }
    
    Tensor::from_vec(positions, &[1, height * width, 2], device)
}

// Helper for 1D positions
pub fn get_1d_positions(seq_len: usize, device: &Device) -> Result<Tensor> {
    let positions: Vec<i32> = (0..seq_len as i32).collect();
    Tensor::from_vec(positions, &[1, seq_len], device)
}
