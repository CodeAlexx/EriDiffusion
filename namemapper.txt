rust//! Diagnostic tool to inspect tensor names in safetensors files

use safetensors::SafeTensors;
use std::fs::File;
use std::io::Read;
use anyhow::Result;

pub fn inspect_safetensors_keys(path: &str) -> Result<Vec<String>> {
    let mut file = File::open(path)?;
    let mut buffer = Vec::new();
    file.read_to_end(&mut buffer)?;
    
    let tensors = SafeTensors::deserialize(&buffer)?;
    let mut keys: Vec<String> = tensors.names().into_iter().collect();
    keys.sort();
    
    Ok(keys)
}

rust//! Updated Flux model loading with proper tensor mapping

use candle_core::{Device, DType, Tensor};
use safetensors::SafeTensors;
use std::collections::HashMap;
use anyhow::Result;

pub struct FluxWeightLoader {
    tensors: HashMap<String, Tensor>,
    device: Device,
    dtype: DType,
}

impl FluxWeightLoader {
    pub fn new(safetensors_data: &[u8], device: Device, dtype: DType) -> Result<Self> {
        let st = SafeTensors::deserialize(safetensors_data)?;
        let mut tensors = HashMap::new();
        
        for (name, tensor_view) in st.tensors() {
            let tensor = tensor_view.load(&device)?
                .to_dtype(dtype)?;
            tensors.insert(name.to_string(), tensor);
        }
        
        Ok(Self { tensors, device, dtype })
    }
    
    pub fn get_tensor(&self, name: &str) -> Result<&Tensor> {
        // First try exact match
        if let Some(tensor) = self.tensors.get(name) {
            return Ok(tensor);
        }
        
        // Try common variations
        let variations = vec![
            name.to_string(),
            name.replace("fc1", "w1"),
            name.replace("fc2", "w2"),
            name.replace("w1", "fc1"),
            name.replace("w2", "fc2"),
            name.replace("to_out.0", "to_out_0"),
            name.replace("to_out_0", "to_out.0"),
        ];
        
        for variant in variations {
            if let Some(tensor) = self.tensors.get(&variant) {
                println!("Mapped {} -> {}", name, variant);
                return Ok(tensor);
            }
        }
        
        // If still not found, print available keys that might match
        let prefix = name.split('.').next().unwrap_or("");
        let similar_keys: Vec<&String> = self.tensors.keys()
            .filter(|k| k.starts_with(prefix))
            .take(5)
            .collect();
            
        anyhow::bail!(
            "Cannot find tensor: {}\nSimilar keys: {:?}\nTotal keys: {}",
            name, similar_keys, self.tensors.len()
        )
    }
    
    pub fn has_tensor(&self, name: &str) -> bool {
        self.tensors.contains_key(name) ||
        self.tensors.contains_key(&name.replace("fc1", "w1")) ||
        self.tensors.contains_key(&name.replace("fc2", "w2"))
    }
    
    pub fn list_keys(&self) -> Vec<&str> {
        let mut keys: Vec<&str> = self.tensors.keys().map(|s| s.as_str()).collect();
        keys.sort();
        keys
    }
}
