Option 3: Write a Tensor Remapper
ruststruct TensorRemapper {
    checkpoint: HashMap<String, Tensor>,
    mappings: HashMap<String, String>,
}

impl TensorRemapper {
    fn load_with_fallbacks(&self, model_path: &str) -> Result<Tensor> {
        // Try original path, then mappings, then synthesize
        self.checkpoint.get(model_path)
            .or_else(|| self.mappings.get(model_path)
                .and_then(|p| self.checkpoint.get(p)))
            .or_else(|| self.synthesize_tensor(model_path))
            .ok_or_else(|| anyhow!("No tensor for {}", model_path))
    }
}
