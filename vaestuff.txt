use candle_core::{Device, DType, Tensor, Shape};

/// Diagnose VAE tensor shapes
pub fn diagnose_vae_shapes(vae_path: &str) -> Result<()> {
    use safetensors::SafeTensors;
    use std::fs::File;
    use std::io::Read;
    
    let mut file = File::open(vae_path)?;
    let mut buffer = Vec::new();
    file.read_to_end(&mut buffer)?;
    
    let tensors = SafeTensors::deserialize(&buffer)?;
    
    println!("=== VAE Tensor Shapes ===");
    
    let mut keys: Vec<String> = tensors.names().into_iter().collect();
    keys.sort();
    
    // Group by encoder/decoder
    for key in keys.iter() {
        if key.contains("encoder") || key.contains("decoder") {
            let shape = tensors.tensor(key)?.shape();
            println!("{}: {:?}", key, shape);
        }
    }
    
    // Look for specific layers that might mismatch
    println!("\n=== Key Layer Shapes ===");
    let important_layers = [
        "encoder.conv_in",
        "encoder.conv_out", 
        "decoder.conv_in",
        "decoder.conv_out",
        "quant_conv",
        "post_quant_conv",
    ];
    
    for layer_prefix in important_layers {
        for key in keys.iter() {
            if key.contains(layer_prefix) {
                let shape = tensors.tensor(key)?.shape();
                println!("{}: {:?}", key, shape);
            }
        }
    }
    
    Ok(())
}

/// Check if tensor shapes match expected
pub fn validate_vae_tensor_shape(
    tensor_name: &str,
    actual_shape: &[usize],
    expected_shape: &[usize],
) -> Result<()> {
    if actual_shape != expected_shape {
        anyhow::bail!(
            "VAE tensor shape mismatch for {}:\n  Expected: {:?}\n  Actual: {:?}",
            tensor_name, expected_shape, actual_shape
        );
    }
    Ok(())
}

/// Adapter to handle shape mismatches
pub struct VAEShapeAdapter {
    expected_latent_channels: usize,
    actual_latent_channels: usize,
}

impl VAEShapeAdapter {
    pub fn new(expected: usize, actual: usize) -> Self {
        Self {
            expected_latent_channels: expected,
            actual_latent_channels: actual,
        }
    }
    
    pub fn adapt_encoder_output(&self, x: &Tensor) -> Result<Tensor> {
        if self.expected_latent_channels == self.actual_latent_channels {
            return Ok(x.clone());
        }
        
        // If shape mismatch, we might need to add/remove channels
        let shape = x.shape();
        if shape.dims()[1] != self.actual_latent_channels {
            anyhow::bail!(
                "Unexpected encoder output channels: {} (expected {})",
                shape.dims()[1], self.actual_latent_channels
            );
        }
        
        // Adapt to expected channels (this is a simplified version)
        // In practice, you might need a learned projection
        if self.actual_latent_channels > self.expected_latent_channels {
            // Take first N channels
            x.narrow(1, 0, self.expected_latent_channels)
        } else {
            // Pad with zeros
            let padding = self.expected_latent_channels - self.actual_latent_channels;
            let zeros = Tensor::zeros(
                (shape.dims()[0], padding, shape.dims()[2], shape.dims()[3]),
                x.dtype(),
                x.device(),
            )?;
            Tensor::cat(&[x, &zeros], 1)
        }
    }
}
